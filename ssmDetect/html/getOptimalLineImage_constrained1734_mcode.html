1function [a, sel_xy] = getOptimalLineImage_constrained(xy, ~, ptsinit, delta)
 2%#codegen
 3
 4
 5a = zeros(3,1);
 6
 7
 8sel_xy = [] ;
 9[U,~,~]=svd(cov(ptsinit')) ;
10
11a(1:2) = U(:,2) ;
12a(3) = -[a(1), a(2)]*mean(ptsinit,2) ;
13
14min_val = 1e-10 ;
15a0 = a;
16sigma = delta ; %[delta, delta/4 ] ;
17for i = 1 : length(sigma)
18    for j = 1 : 5
19        
20        rr = abs(a(1)*xy(1,:) + a(2)*xy(2,:) + a(3))/sigma(i) ;
21        zerovls = rr>2 ;
22        w = exp( -0.5*rr.^2 ) ;
23        
24        w(zerovls) = 0 ;
25        w = w/sum(w) ;
26        
27        %mnxy = sum(bsxfun(@times, xy, w),2) ;  
28        xyw=zeros(size(xy));
29        for k=1:size(xy,1),
30            xyw(k,:)= xy(k,:).*w;
31        end
32        mnxy=sum(xyw,2);
33        
34        %wd = bsxfun(@times, bsxfun(@minus,xy,mnxy), sqrt(w)) ;
35        sub=zeros(size(xy));
36        for k=1:size(xy,2),
37            sub(:,k)= xy(:,k)-mnxy;
38        end
39        
40        wd=zeros(size(sub));
41        for k=1:size(sub,1),
42            wd(k,:)= sub(k,:).*sqrt(w);
43        end
44        
45        
46        C = wd*wd' ;
47        [U,~,~] = svd(C);
48        a(1:2) = U(:,2) ;  a(3) = -[a(1), a(2)]*mnxy ;
49        
50        if mean(abs(a0-a)) < min_val
51            break ;
52        end
53        a0 = a ;
54        
55    end
56end
57if nargout > 1
58    sel_xy = xy(:,zerovls==0) ;
59end
60end
