 89function objs_out = pruneobjs( bbs, parts, selected )
 90%#codegen
 91
 92objs_out = Object(numel(selected));
 93
 94for i = 1 : length(selected)
 95    objs_out(i).bounding_box =  bbs(i,:) ;
 96    objs_out(i).area = prod(bbs(i,[3,4])) ;
 97    objs_out(i).parts = parts(selected(i).idx(:)) ;
 98end
 99end
