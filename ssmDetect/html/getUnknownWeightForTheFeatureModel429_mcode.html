1function p_unknown = getUnknownWeightForTheFeatureModel(type_colorspace, sizeMask, use_uniform_component) 
 2 %#codegen
 3 
 4p_unknown=0;  %zato da je output predvidljiv
 5 
 6if ~use_uniform_component
 7    return ;
 8end
 9
10switch(type_colorspace)
11    case Colorspace.hsv
12        p_unknown = 1/prod(sizeMask) ; 1e-5 ; 0.001 ;
13    case Colorspace.rgb
14        p_unknown = 1/(prod(sizeMask) *255^3) *0.001; 1e-5 ; 0.001 ;
15    case Colorspace.ycrcb
16        p_unknown = 0.001*1/(prod(sizeMask) *10988544) ; %
17    case Colorspace.lab
18        p_unknown =10^(-2) *  1 / ( prod(sizeMask) *255^3  ) ;%
19    case Colorspace.ycrs
20        p_unknown = 0.01*1/(prod(sizeMask) *49056) ; 
21    otherwise
22        error('Invalid type colorspace.');  
23end
24end
